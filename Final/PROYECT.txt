create database ProyectFinal

CREATE TABLE USUARIO(
IDUSU INT IDENTITY(1,1) PRIMARY KEY,
NOMUSU VARCHAR(50),
CLAUSU VARCHAR(50),

);

CREATE TABLE PACIENTE(
IDPA INT IDENTITY(1,1) PRIMARY KEY,
NOMPA VARCHAR(50),
APEPA VARCHAR(50),
CORPA VARCHAR(50),
DNIPA INT(12),
CELPA INT(12)

);

CREATE TABLE DOCTOR(
IDDOC INT IDENTITY(1,1) PRIMARY KEY,
NOMDOC VARCHAR(50),
APEDOC VARCHAR(50),
ESPDOC VARCHAR(50),
CELDOC INT(12)
);


CREATE TABLE CITA(
IDCI INT IDENTITY(1,1) PRIMARY KEY,
IDDOC INT NOT NULL REFERENCES DOCTOR,
IDPA INT NOT NULL REFERENCES PACIENTE,
FECIN DATE NOT NULL,
FECFI DATE NOT NULL
);
